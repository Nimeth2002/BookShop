    private void btnRegisterActionPerformed(java.awt.event.ActionEvent evt) {                                            
        FileWriter writer = null;
    try {
// TODO add your handling code here:
        String Name = txtname.getText();
        String UserID = txtUserID.getText();
        ;
        String Address = txtAddress.getText();
        String Password = txtPassword.getText();
        String ConfirmPassword = txtConfirmpassword.getText();
        String Usertype = cmbUserType.getSelectedItem().toString();
        String Record = null;
        Record = Name + "" + UserID + "" + Address + "" + Password + "" + ConfirmPassword + "" + Usertype + "";
        writer = new FileWriter (myFile,true);
        BufferedWriter buffer = new BufferedWriter(writer);
       
        buffer.write(Record);
        buffer.newLine();
        buffer.close();
       
        JOptionPane.showMessageDialog(null,"Successfully Registered");
    } catch (IOException ex) {
        JOptionPane.showMessageDialog(null, "An Error Occurres"+ex);
    } finally {
        try {
            writer.close();
        } catch (IOException ex) {
            Logger.getLogger(NewJFrameRegistration.class.getName()).log(Level.SEVERE, null, ex);
        }
    }
        
        
       
        
        
    }
