    private void btnSaveActionPerformed(java.awt.event.ActionEvent evt) {                                        
        FileWriter Writer = null;
    try {
        // TODO add your handling code here:
        String Bookname = txtBookname.getText();
        String Booktitle = txtBooktitle.getText();
        String Bookcatogary = txtBookCatogary.getText();
        String Authorname = txtAuthorname.getText();
        String Publishername = txtPublishername.getText();
        String Price = txtPrice.getText();
        String Record = null;
        Record = Bookname + "" + Booktitle + "" + Bookcatogary + "" + Authorname + "" + Publishername + "" + Price + "" ;
        Writer = new FileWriter (myFile,true);
        BufferedWriter buffer = new BufferedWriter(Writer);
        buffer.write(Record);
        buffer.newLine();
        buffer.close();
        JOptionPane.showMessageDialog(null,"Successfully Wrote into the File");
    } catch (IOException ex) {
        JOptionPane.showMessageDialog(null, "An Error Occurres"+ex);
    } finally {
        try {
            Writer.close();
        } catch (IOException ex) {
            Logger.getLogger(CityBookshop.class.getName()).log(Level.SEVERE, null, ex);
        }
    }
        
        
    }                                       

    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        try
        {
            boolean found = true;
            String [] words = null;
            FileReader fileR = new FileReader(myFile);
            BufferedReader br = new BufferedReader(fileR);
            
            String FindLine;
            String input = txtBookname.getText();
            
            outerLoop:
            while ((FindLine=br.readLine())!=null)
            {
                words = FindLine.split("");
                for (String word : words)
                {
                    if(word.equals(input))
                    {
                        found = true;
                        System.out.println(FindLine);
                        
                        break outerLoop;
                    }
                    else
                    {
                        found = false;
                    }
                }   
            }
            if(found)
            {
                JOptionPane.showMessageDialog(null,"Value found"+"\n"+FindLine);
            }
            else
            {
                JOptionPane.showMessageDialog(null,"Value not found");
            }
            fileR.close();
            br.close();
          } catch (IOException ex){
              System.out.println("An error occurred searching"+ex);
        Logger.getLogger(CityBookshop.class.getName()).log(Level.SEVERE, null, ex);
    }
// TODO add your handling code here:
        
            
            
    }
